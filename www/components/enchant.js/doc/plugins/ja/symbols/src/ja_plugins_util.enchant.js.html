 20 /**
 21  * @scope enchant.util.MutableText.prototype
 22  * @type {*}
 23  */
 24 enchant.util.MutableText = enchant.Class.create(enchant.Sprite, {
 25     /**
 26      * ビットマップフォントを用いたラベルクラス
 27      * (参考: draw.text.js http://d.hatena.ne.jp/nakamura001/20110430/1304181043)
 28      * enchant.js 添付素材の font*.png が利用可能
 29      * @constructs
 30      * @param posX
 31      * @param posY
 32      * @param width
 33      */
 34     initialize: function(posX, posY, width) {
 35         enchant.Sprite.call(this, 0, 0);
 36         this.fontSize = 16;
 37         this.widthItemNum = 16;
 38         this.x = posX;
 39         this.y = posY;
 40         this.text = '';
 41         if (arguments[2]) {
 42             this.row = Math.floor(arguments[2] / this.fontSize);
 43         }
 44     },
 45     /**
 46      * ラベルの内容を書き換える関数
 47      * @param txt
 48      */
 49     setText: function(txt) {
 50         var i, x, y, wNum, charCode, charPos;
 51         this._text = txt;
 52         if (!this.returnLength) {
 53             this.width = Math.min(this.fontSize * this._text.length, enchant.Game.instance.width);
 54         } else {
 55             this.width = Math.min(this.returnLength * this.fontSize, enchant.Game.instance.width);
 56         }
 57         this.height = this.fontSize * (Math.ceil(this._text.length / this.row) || 1);
 58         this.image = new enchant.Surface(this.width, this.height);
 59         this.image.context.clearRect(0, 0, this.width, this.height);
 60         for (i = 0; i < txt.length; i++) {
 61             charCode = txt.charCodeAt(i);
 62             if (charCode >= 32 && charCode <= 127) {
 63                 charPos = charCode - 32;
 64             } else {
 65                 charPos = 0;
 66             }
 67             x = charPos % this.widthItemNum;
 68             y = (charPos / this.widthItemNum) | 0;
 69             this.image.draw(enchant.Game.instance.assets['font.png'],
 70                 x * this.fontSize, y * this.fontSize, this.fontSize, this.fontSize,
 71                 (i % this.row) * this.fontSize, ((i / this.row) | 0) * this.fontSize, this.fontSize, this.fontSize);
 72         }
 73     },
 74     /**
 75      * ラベルの内容
 76      * @type {String}
 77      */
 78     text: {
 79         get: function() {
 80             return this._text;
 81         },
 82         set: function(txt) {
 83             this.setText(txt);
 84         }
 85     },
 86     /**
 87      * @type {Number}
 88      */
 89     row: {
 90         get: function() {
 91             return this.returnLength || this.width / this.fontSize;
 92         },
 93         set: function(row) {
 94             this.returnLength = row;
 95             this.text = this.text;
 96         }
 97     }
 98 });

100 /**
101  * @scope enchant.util.ScoreLabel.prototype
102  * @type {*}
103  */
104 enchant.util.ScoreLabel = enchant.Class.create(enchant.util.MutableText, {
105     /**
106      * スコアを表示するラベル。
107      * 画像フォントクラス (MutableText) を使って表示する。
108      * @constructs
109      * @param x
110      * @param y
111      */
112     initialize: function(x, y) {
113         enchant.util.MutableText.call(this, 0, 0);
114         switch (arguments.length) {
115             case 2:
116                 this.y = y;
117                 this.x = x;
118                 break;
119             case 1:
120                 this.x = x;
121                 break;
122             default:
123                 break;
124         }
125         this._score = 0;
126         this._current = 0;
127         this.easing = 2.5;
128         this.text = this.label = 'SCORE:';
129         this.addEventListener('enterframe', function() {
130             if (this.easing === 0) {
131                 this.text = this.label + (this._current = this._score);
132             } else {
133                 this._current += Math.ceil((this._score - this._current) / this.easing);
134                 this.text = this.label + this._current;
135             }
136         });
137     },
138     /**
139      * スコア
140      * @type {Number}
141      */
142     score: {
143         get: function() {
144             return this._score;
145         },
146         set: function(newscore) {
147             this._score = newscore;
148         }
149     }
150 });

155  */
156 enchant.util.TimeLabel = enchant.Class.create(enchant.util.MutableText, {
157     /**
158      * 残り時間などのタイムを表示するラベル
159      * @constructs
160      * @param x
161      * @param y
162      * @param counttype
163      */
164     initialize: function(x, y, counttype) {
165         enchant.util.MutableText.call(this, 0, 0);
166         switch (arguments.length) {
167             case 3:
168             case 2:
169                 this.y = y;
170                 this.x = x;
171                 break;
172             case 1:
173                 this.x = x;
174                 break;
175             default:
176                 break;
177         }
178         this._time = 0;
179         this._count = 1;// この数を毎フレーム每に足して上げ下げを制御する
180         if (counttype === 'countdown') {
181             this._count = -1;
182         }
183         this.text = this.label = 'TIME:';
184         this.addEventListener('enterframe', function() {
185             this._time += this._count;
186             this.text = this.label + (this._time / enchant.Game.instance.fps).toFixed(2);
187         });
188     },
189     /**
190      * 残り時間
191      * @type {Number}
192      */
193     time: {
194         get: function() {
195             return Math.floor(this._time / enchant.Game.instance.fps);
196         },
197         set: function(newtime) {
198             this._time = newtime * enchant.Game.instance.fps;
199         }
200     }
201 });

203 /**
204  * @type {*}
205  * @scope enchant.util.LifeLabel.prototype
206  */
207 enchant.util.LifeLabel = enchant.Class.create(enchant.Group, {
208     /**
209      * ライフを表示する専用のラベル
210      * icon0.gif 内のハートの画像を用いる
211      * @constructs
212      * @param x
213      * @param y
214      * @param maxlife
215      */
216     initialize: function(x, y, maxlife) {
217         enchant.Group.call(this);
218         this.x = x || 0;
219         this.y = y || 0;
220         this._maxlife = maxlife || 9;
221         this._life = 0;
222         this.label = new enchant.MutableText(0, 0, 80, 'LIFE:');
223         this.addChild(this.label);
224         this.heart = [];
225         for (var i = 0; i < this._maxlife; i++) {
226             this.heart[i] = new enchant.Sprite(16, 16);
227             this.heart[i].image = enchant.Game.instance.assets['icon0.gif'];
228             this.heart[i].x = this.label.width + i * 16;
229             this.heart[i].y = -3;
230             this.heart[i].frame = 11;
231             this.addChild(this.heart[i]);
232         }
233     },
234     /**
235      * 残りライフの数
236      * @type {Number}
237      */
238     life: {
239         get: function() {
240             return this._life;
241         },
242         set: function(newlife) {
243             this._life = newlife;
244             if (this._maxlife < newlife) {
245                 this._life = this._maxlife;
246             }
247             for (var i = 0; i < this._maxlife; i++) {
248                 this.heart[i].visible = (i <= newlife - 1);
249             }
250         }
251     }
252 });

254 /**
255  * @scope enchant.util.Bar
256  * @type {*}
257  */
258 enchant.util.Bar = enchant.Class.create(enchant.Sprite, {
259     /**
260      * イージング付きのバークラス
261      * @constructs
262      * @param x
263      * @param y
264      */
265     initialize: function(x, y) {
266         enchant.Sprite.call(this, 1, 16);
267         this.image = new enchant.Surface(1, 16);// Null用
268         this.image.context.fillColor = 'RGB(0, 0, 256)';
269         this.image.context.fillRect(0, 0, 1, 16);
270         this._direction = 'right';
271         this._origin = 0;
272         this._maxvalue = enchant.Game.instance.width;
273         this._lastvalue = 0;
274         this.value = 0;
275         this.easing = 5;
276         switch (arguments.length) {
277             case 2:
278                 this.y = y;
279                 this.x = x;
280                 this._origin = x;
281                 break;
282             case 1:
283                 this.x = x;
284                 this._origin = x;
285                 break;
286             default:
287                 break;
288         }
289         this.addEventListener('enterframe', function() {
290             if (this.value < 0) {
291                 this.value = 0;
292             }
293             this._lastvalue += (this.value - this._lastvalue) / this.easing;
294             if (Math.abs(this._lastvalue - this.value) < 1.3) {
295                 this._lastvalue = this.value;
296             }
297             this.width = (this._lastvalue) | 0;
298             if (this.width > this._maxvalue) {
299                 this.width = this._maxvalue;
300             }
301             if (this._direction === 'left') {
302                 this._x = this._origin - this.width;
303             } else {
304                 this._x = this._origin;
305             }
306             this._updateCoordinate();
307         });
308     },
309     /**
310      * バーの向き ('right' or 'left')
311      * @default 'right'
312      * @type {String}
313      */
314     direction: {
315         get: function() {
316             return this._direction;
317         },
318         set: function(newdirection) {
319             if (newdirection !== 'right' && newdirection !== 'left') {
320                 // ignore
321             } else {
322                 this._direction = newdirection;
323             }
324         }
325     },
326     /**
327      * x 座標
328      * @type {Number}
329      */
330     x: {
331         get: function() {
332             return this._origin;
333         },
334         set: function(x) {
335             this._x = x;
336             this._origin = x;
337             this._updateCoordinate();
338         }
339     },
340     /**
341      * @type {Number}
342      */
343     maxvalue: {
344         get: function() {
345             return this._maxvalue;
346         },
347         set: function(val) {
348             this._maxvalue = val;
349         }
350     }
351 });

353 /**
354  * @scope enchant.util.VirtualMap.prototype
355  */
356 enchant.util.VirtualMap = enchant.Class.create(enchant.Group, {
357     /**
358      * マップライクな Group
359      * addChildで Sprite 等を追加すると、自動的に mx, my プロパティが追加され、
360      * VirtualMap内での座標で Sprite を操作できる
361      *
362      * 使い方
363      * //20 x 20 メッシュの縦横320ピクセルの盤を作り、その上に16 x 16の駒を8つ並べる
364      * var board = new VirtualMap(20, 20);
365      * board.width = 320;
366      * board.height = 320;
367      * for(var i=0; i<8; i++){
368      *     var piece = new Sprite(16, 16);
369      *     piece.image = game.assets['icon0.gif'];
370      *     board.addChild(piece);
371      *     piece.mx = i + 3;
372      *     piece.my = 16;
373      * }
374      * game.rootScene.addChild(board);
375      *
376      * @param meshWidth
377      * @param meshHeight
378      * @constructs
379      */
380     initialize: function(meshWidth, meshHeight) {
381         enchant.Group.call(this);
382         this.meshWidth = meshWidth || 16;
383         this.meshHeight = meshHeight || 16;
384     },
385     /**
386      * VirtualMap にオブジェクトを追加する (自動的にバインドされる)
387      * @param obj
388      */
389     addChild: function(obj) {
390         enchant.Group.prototype.addChild.call(this, obj);
391         this.bind(obj);
392     },
393     /**
394      * VirtualMap にオブジェクトを追加する
395      * reference で指定したオブジェクトより前に追加される (自動的にバインドされる)。
396      * @param obj
397      * @param reference
398      */
399     insertBefore: function(obj, reference) {
400         enchant.Group.prototype.insertBefore.call(this, obj, reference);
401         this.bind(obj);
402     },
403     /**
404      * オブジェクトを VirtualMap にバインドする。
405      * バインドされたオブジェクトはメッシュ座標 mx, my プロパティを持ち、これを操作することで
406      * VirtualMap の中を移動させることができる。
407      * @param obj
408      */
409     bind: function(obj) {
410         Object.defineProperties(obj, {
411             "mx": {
412                 get: function() {
413                     return Math.floor(this.x / this.parentNode.meshWidth);
414                 },
415                 set: function(arg) {
416                     this.x = Math.floor(arg * this.parentNode.meshWidth);
417                 }
418             },
419             "my": {
420                 get: function() {
421                     return Math.floor(this.y / this.parentNode.meshHeight);
422                 },
423                 set: function(arg) {
424                     this.y = Math.floor(arg * this.parentNode.meshWidth);
425                 }
426             }
427         });
428         obj.mx = 0;
429         obj.my = 0;
430     }
431 });
